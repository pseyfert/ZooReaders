66 bool ZooSTChannelID::isTop() const
67 {
68  if (isIT()) return 4 == region();
69  // is TT
70  if (1 == (region() & 1)) {
71  // region 1 or 3
72  return (sector() & 3) > 2;
73  }
74  // region 2
75  if (1 == station()) return ((sector() - 1) % 6) > 2;
76  // region 2 in TTB
77  if (sector() < 12 || sector() > 22)
78  return std::min(sector(), sector() - 22) > 2;
79  // middle of region 2 in TTB
80  return ((sector() - 5) % 6) > 2;
