68  template<typename U, typename P, unsigned N>
69  typename Unit<U, P, N>::packed Unit<U, P, N>::doPack(
70  typename Unit<U, P, N>::unpacked u)
71  {
72  // encode NaNs specially
73  if (UNLIKELY(std::isnan(u)))
74  return std::numeric_limits<packed>::min();
75  // proper rounding (round to even: what you would expect,
76  // except for 0.5, where we round either up or down depending on
77  // the LSB of the rounded integer; this is correct on average:
78  // 0.5 is a tie, and we choose both of the possible outcomes
79  // exactly half of the time (assuming that the LSB of the rounded
80  // integer is random which it should be))
81  unpacked fu = std::floor(u);
82  if (u - fu > 0.5) u += unpacked(1);
83  else if (u - fu == 0.5 && packed(fu) & 1) u += unpacked(1);
84  // saturate
85  if (UNLIKELY(unpacked(std::numeric_limits<packed>::max()) < u))
86  return std::numeric_limits<packed>::max();
87  if (UNLIKELY(-unpacked(std::numeric_limits<packed>::max()) > u))
88  return -std::numeric_limits<packed>::max();
89  return packed(std::floor(u));
90  }

105  {

115  {
